- sudo apt-get install mongodb
- start mongo sudo /etc/init.d/mongodb start or sudo service mongodb start

- connect to mongo db server, "mongo"
- create database, "use pingpong"
- create api file structure src/controllers, src/routes
- install express, npm install express
- create src/config.js
